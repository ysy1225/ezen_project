create table notice(
num number primary key,
writer varchar2(30),
subject varchar2(500),
notice_date varchar2(10),
readcount number default 0,
content varchar2(4000));

create sequence notice_num;

create table tutor(
tutorNum number primary key,
tutorEmail varchar2(50),
tutorSubject varchar2(50),
location varchar2(100),
time number not null,
min number not null,
max number not null,
price number not null,
tutorContent varchar2(4000),

create sequence tutor_num;



create table users(
userNum number primary key,
userName varchar2(30) not null,
userEmail varchar2(100) not null,
userPhone varchar2(50) not null,
userPw1 varchar2(40) not null,
userPw2 varchar2(40) not null,
userRegdate date default sysdate
userLevel number default 0);		

create sequence users_seq;
